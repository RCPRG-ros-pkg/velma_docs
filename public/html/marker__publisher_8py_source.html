53  def publishSinglePointMarker(self, pt, i, r=1, g=0, b=0, a=1, namespace='default', frame_id='torso_base', m_type=Marker.CUBE, scale=Vector3(0.005, 0.005, 0.005), T=None):
54  m = MarkerArray()
55  marker = Marker()
56  marker.header.frame_id = frame_id
57  marker.header.stamp = rospy.Time.now()
58  marker.ns = namespace
59  marker.id = i
60  marker.type = m_type
61  marker.action = Marker.ADD
62  if T != None:
63  point = T*pt
64  q = T.M.GetQuaternion()
65  marker.pose = Pose( Point(point.x(),point.y(),point.z()), Quaternion(q[0],q[1],q[2],q[3]) )
66  else:
67  marker.pose = Pose( Point(pt.x(),pt.y(),pt.z()), Quaternion(0,0,0,1) )
68  marker.scale = scale
69  marker.color = ColorRGBA(r,g,b,a)
70  m.markers.append(marker)
71  self._pub_marker.publish(m)
72  return i+1
