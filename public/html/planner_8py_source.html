39 def qMapToConstraints(q_map, tolerance=0.01, group=None):
40  """!
41  This function converts dictionary of joint positions to moveit_msgs.Constraints structure.
42 
43  @param q_map dictionary: A dictionary {name:position} of desired joint positions.
44  @param tolerance float: Tolerance of goal position.
45  @param group list of strings: Names of active joints in planning group.
46  Only constraints for these joints should be generated.
47  @return Returns filled moveit_msgs.Constraints structure.
48  """
49  result = Constraints()
50  for joint_name in q_map:
51  if group != None and not joint_name in group:
52  continue
53  constraint = JointConstraint()
54  constraint.joint_name = joint_name
55  constraint.position = q_map[ joint_name ]
56  constraint.tolerance_above = tolerance
57  constraint.tolerance_below = tolerance
58  constraint.weight = 1.0
59  result.joint_constraints.append( constraint )
60  return result

87  def getOctomap(self, timeout_s=None):
88  """!
89  Get current octomap from octomap server.
90 
91  @param timeout_s float: Timeout in seconds.
92 
93  @return Octomap object of type octomap_msgs.msg.Octomap, or None if timed out.
94  """
95  if timeout_s == None:
96  with self._lock:
97  if self._octomap:
98  return copy.copy(self._octomap)
99  return None
100  else:
101  time_start = time.time()
102  while not rospy.is_shutdown():
103  rospy.sleep(0.1)
104  with self._lock:
105  if self._octomap:
106  return copy.copy(self._octomap)
107  time_now = time.time()
108  if timeout_s and (time_now-time_start) > timeout_s:
109  return None
110  return None
