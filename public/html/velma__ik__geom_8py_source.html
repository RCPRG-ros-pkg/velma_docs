81  a = self.__arm_len_a
82  b = self.__arm_len_b
83  acos_val = (a**2 + b**2 - dist**2) / (2*a*b)

92  def __heronArea(self, a, b, c):
93  s = (a+b+c)/2.0
94  val = s*(s-a)*(s-b)*(s-c)
95  if val < 0:
96  raise Exception('Square root of negative number: {} = {}*({}-{})*({}-{})*({}-{})'.format(val, s, s, a, s, b, s, c))
97  return math.sqrt(val)

99  def __calculateIkPart1(self, T_A0_A6d, elbow_circle_angle, flip_shoulder, flip_elbow):
100 
101  pt_shoulder = PyKDL.Vector(0, 0, self.__D)
102  dir_vec = T_A0_A6d.p - pt_shoulder;
103  dist = dir_vec.Norm()
104  q3 = self.__calcQ3(dist)
105 
106  if q3 is None:
107  return None, None, None, None
108 
109  area = self.__heronArea(dist, self.__arm_len_a, self.__arm_len_b)
110  height = area / (0.5*dist)
111  dist_a = math.sqrt( self.__arm_len_a**2 - height**2 )
112 
113  # The elbow circle angle is calculated wrt the axis of the first joint of the arm.
114  # This is the parameter for IK that deals with redundancy problem.
115 
116  shoulder_vec = pt_shoulder - PyKDL.Vector()
117  nz = dir_vec * 1.0
118  nx = shoulder_vec * 1.0
119  ny = nz*nx
120  nx = ny*nz
121  nx.Normalize()
122  ny.Normalize()
123  nz.Normalize()
124  shoulder_frame = PyKDL.Frame( PyKDL.Rotation(nx, ny, nz), pt_shoulder )
125  shoulder_frame_rot = shoulder_frame * PyKDL.Frame( PyKDL.Rotation.RotZ(elbow_circle_angle), PyKDL.Vector() )
126  elbow_pt = shoulder_frame_rot * PyKDL.Vector(height, 0, dist_a)
127  elbow_vec = elbow_pt - pt_shoulder
128  q0 = math.atan2(-elbow_pt.y(), -elbow_pt.x())
129 
130  q1 = -math.atan2(math.sqrt(elbow_vec.x()**2 + elbow_vec.y()**2), elbow_vec.z())
131 
132  # Calculate q2:
133  # Project dir_vec to frame of right_arm_2_link, and compute atan2(-dir_vec.y(), -dir_vec.x())
134 
135  nx = shoulder_vec * 1.0
136  nz = elbow_vec * 1.0
137  ny = nz*nx
138  nx = ny*nz
139  nx.Normalize()
140  ny.Normalize()
141  nz.Normalize()
142  T_A0_A2 = PyKDL.Frame( PyKDL.Rotation(nx, ny, nz), pt_shoulder )
143  dir_vec_A0 = dir_vec
144  dir_vec_A2 = T_A0_A2.Inverse().M * dir_vec_A0
145  q2 = math.atan2(-dir_vec_A2.y(), -dir_vec_A2.x())
146 
147  # There are two alternative angles for q3
148  if flip_elbow:
149  q3 = -q3
150  q0 = math.pi+q0
151  q1 = -q1
152 
153  if flip_shoulder:
154  q0 = math.pi+q0
155  q1 = -q1
156  q2 = math.pi+q2
157 
158  # These points are saved for debugging
159  self.__debug['pt_shoulder'] = pt_shoulder
160  self.__debug['elbow_pt'] = elbow_pt
161  #self.__debug['ee_pt'] = ee_pt
162 
163  # There are two alternative angles for q3
164  if flip_elbow:
165  q0 = math.pi + q0
166  q1 = -q1
167  q3 = -q3
168 
169  if flip_shoulder:
170  q0 = math.pi + q0
171  q1 = -q1
172  q2 = math.pi + q2
173 
174  q0 = wrapAngle(q0)
175  q1 = wrapAngle(q1)
176  q2 = wrapAngle(q2)
177  q3 = wrapAngle(q3)
178  if q0 < self.m_lim_lo[0] or q0 > self.m_lim_up[0] or\
179  q1 < self.m_lim_lo[1] or q1 > self.m_lim_up[1] or\
180  q2 < self.m_lim_lo[2] or q2 > self.m_lim_up[2] or\
181  q3 < self.m_lim_lo[3] or q3 > self.m_lim_up[3]:
182  return None, None, None, None
183 
184  # else
185  return q0, q1, q2, q3

239  q5 = wrapAngle(q5)
240  q6 = wrapAngle(q6)

242  if q4 < self.m_lim_lo[4] or q4 > self.m_lim_up[4] or\
243  q5 < self.m_lim_lo[5] or q5 > self.m_lim_up[5] or\

250  """!
251  Calculate inverse kinematics (IK) for Kuka LWR 4+.
252

254  (the first link).
255  @param elbow_circle_angle float: IK parameter.
256  @param flip_shoulder bool: IK parameter.
257  @param flip_elbow bool: IK parameter.

259 
260  @return 7-tuple: arm configuration or a tuple of Nones, if the solution does not exist.
261  """
262 
263  self.__debug = {}
264  T_A6_A7 = PyKDL.Frame(PyKDL.Rotation(), PyKDL.Vector(0, 0, self.__A))
265  T_A7_A6 = T_A6_A7.Inverse()

267  self.__debug['T_A0_A6d'] = T_A0_A6d

269  if q0 is None:

271  # else

273  if q4 is None:
274  return None, None, None, None, None, None, None
275  # else
276  return q0, q1, q2, q3, q4, q5, q6
