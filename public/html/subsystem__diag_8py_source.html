49 def parsePredicates(pred_str):
50  pred_list = pred_str.split(",")
51  result = []
52  for cp in pred_list:
53  k_v = cp.split(":")
54  if len(k_v) == 2:
55  pv = SubsystemDiag.PredicateValue()
56  pv.name = k_v[0].strip()
57  value_str = k_v[1].strip()
58  if value_str == 't':
59  pv.value = True
60  elif value_str == 'f':
61  pv.value = False
62  else:
63  raise ValueError("wrong predicate value: '" + value_str + "' for predicate: '" + pv.name + "'")
64  result.append(pv)
65  return result
